610  inline void SparseUpdateChunk(const NDArray &arr) const {
611  CHECK(shape_ == arr.shape_) << "ndarray shape is different from the target";
612  CHECK(dtype_ == arr.dtype_) << "ndarray dtype is different from the target";
613  auto stype = arr.storage_type();
614  CHECK(stype == kCSRStorage || stype == kRowSparseStorage)
615  << "Only to be used with CSR and RSP storage types";
616  // swap shandles between src and dst
617  Storage::Handle shandle_dst = arr.ptr_->shandle;
618  arr.ptr_->shandle = ptr_->shandle;
619  ptr_->shandle = shandle_dst;
620 
621  ptr_->storage_shape = arr.ptr_->storage_shape;
622  ptr_->storage_type = arr.ptr_->storage_type;
623  ptr_->ctx = arr.ptr_->ctx;
624 
625  // swap aux_handles between src and dst
626  size_t aux_idx = 0;
627  CHECK(ptr_->aux_handles.size() == arr.ptr_->aux_handles.size())
628  << "ndarray number of aux_handles is different from target";
629  for (auto &aux_handle : arr.ptr_->aux_handles) {
630  Storage::Handle aux_dst = ptr_->aux_handles[aux_idx];
631  ptr_->aux_handles[aux_idx] = aux_handle;
632  aux_handle = aux_dst;
633  aux_idx++;
634  }
635  ptr_->aux_types = arr.ptr_->aux_types;
636  ptr_->aux_shapes = arr.ptr_->aux_shapes;
637  }
