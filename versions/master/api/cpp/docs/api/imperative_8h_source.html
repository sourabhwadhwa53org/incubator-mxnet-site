206  NumpyShape flag = static_cast<NumpyShape>(is_np_shape);
207  bool old = this->is_np_shape();
208  switch (flag) {
209  case GlobalOn:
210  is_np_shape_global_ = true;
211  is_np_shape_thread_local_ = true;

213  case ThreadLocalOn:
214  is_np_shape_thread_local_ = true;

220  }
