632  inline void SparseUpdateChunk(const NDArray &arr) const {
633  CHECK(shape_ == arr.shape_) << "ndarray shape is different from the target";
634  CHECK(dtype_ == arr.dtype_) << "ndarray dtype is different from the target";
635  auto stype = arr.storage_type();
636  CHECK(stype == kCSRStorage || stype == kRowSparseStorage)
637  << "Only to be used with CSR and RSP storage types";
638  // swap shandles between src and dst
639  Storage::Handle shandle_dst = arr.ptr_->shandle;
640  arr.ptr_->shandle = ptr_->shandle;
641  ptr_->shandle = shandle_dst;
642 
643  ptr_->storage_shape = arr.ptr_->storage_shape;
644  ptr_->storage_type = arr.ptr_->storage_type;
645  ptr_->ctx = arr.ptr_->ctx;
646 
647  // swap aux_handles between src and dst
648  size_t aux_idx = 0;
649  CHECK(ptr_->aux_handles.size() == arr.ptr_->aux_handles.size())
650  << "ndarray number of aux_handles is different from target";
651  for (auto &aux_handle : arr.ptr_->aux_handles) {
652  Storage::Handle aux_dst = ptr_->aux_handles[aux_idx];
653  ptr_->aux_handles[aux_idx] = aux_handle;
654  aux_handle = aux_dst;
655  aux_idx++;
656  }
657  ptr_->aux_types = arr.ptr_->aux_types;
658  ptr_->aux_shapes = arr.ptr_->aux_shapes;
659  }
