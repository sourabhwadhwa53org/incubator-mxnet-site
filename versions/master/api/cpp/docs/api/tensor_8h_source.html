80 template<int dimension>
81 struct Shape {
83  static const int kDimension = dimension;
85  static const int kSubdim = dimension - 1;
87  index_t shape_[kDimension];
89  MSHADOW_XINLINE Shape(void) {}
91  MSHADOW_XINLINE Shape(const Shape<kDimension> &s) {
92  #pragma unroll
93  for (int i = 0; i < kDimension; ++i) {
94  this->shape_[i] = s[i];
95  }
96  }
102  MSHADOW_XINLINE index_t &operator[](int idx) {
103  return shape_[idx];
104  }
110  MSHADOW_XINLINE const index_t &operator[](int idx) const {
111 #pragma GCC diagnostic push
112 #pragma GCC diagnostic ignored "-Warray-bounds"
113  return shape_[idx];
114 #pragma GCC diagnostic pop
115  }
120  MSHADOW_XINLINE bool operator==(const Shape<kDimension> &s) const {
121  #pragma unroll
122  for (int i = 0; i < kDimension; ++i) {
123  if (s.shape_[i] != this->shape_[i]) return false;
124  }
125  return true;
126  }
131  MSHADOW_XINLINE bool operator!=(const Shape<kDimension> &s) const {
132  return !(*this == s);
133  }
138  MSHADOW_XINLINE Shape<1> FlatTo1D(void) const {
139  Shape<1> s;
140  s[0] = this->Size();
141  return s;
142  }
147  MSHADOW_XINLINE Shape<2> FlatTo2D(void) const {
148  Shape<2> s;
149  s.shape_[1] = this->shape_[kDimension - 1];
150  index_t ymax = 1;
151  #pragma unroll
152  for (int i = 0; i < kDimension - 1; ++i) {
153  ymax *= this->shape_[i];
154  }
155  s.shape_[0] = ymax;
156  return s;
157  }
159  MSHADOW_XINLINE index_t Size(void) const {
160  index_t size = this->shape_[0];
161  #pragma unroll
162  for (int i = 1; i < kDimension; ++i) {
163  size *= this->shape_[i];
164  }
165  return size;
166  }
172  MSHADOW_XINLINE index_t ProdShape(int dimstart, int dimend) const {
173  index_t num = 1;
174  #pragma unroll
175  for (int i = dimstart; i < dimend; ++i) {
176  num *= this->shape_[i];
177  }
178  return num;
179  }
184  MSHADOW_XINLINE Shape<kSubdim> SubShape(void) const {
185  Shape<kSubdim> s;
186  // for cuda
187  #pragma unroll
188  for (int i = 0; i < kSubdim; ++i) {
189  s.shape_[i] = this->shape_[i + 1];
190  }
191  return s;
192  }
199  template<int dimstart, int dimend>
200  MSHADOW_XINLINE Shape<dimend - dimstart> Slice(void) const {
201  Shape<dimend - dimstart> s;
202  #pragma unroll
203  for (int i = dimstart; i < dimend; ++i) {
204  s[i - dimstart] = this->shape_[i];
205  }
206  return s;
207  }
209  template<int dim>
210  friend std::ostream &operator<<(std::ostream &os, const Shape<dim> &shape); // NOLINT(*)
212 }; // Shape

284 inline Shape<3> ConvertLayout(const Shape<3>& src, int src_layout, int dst_layout) {
285  Shape<3> dst;
286  switch (src_layout) {
287  case kNCW:
288  dst = src;
289  break;
290  case kNWC:
291  dst[0] = src[0];
292  dst[1] = src[2];
293  dst[2] = src[1];
294  break;
295  default:
296  LOG(FATAL) << "Invalid layout for 3d shape " << src_layout;
297  }
298  switch (dst_layout) {
299  case kNCW:
300  return dst;
301  case kNWC:
302  {
303  index_t tmp = dst[1];
304  dst[1] = dst[2];
305  dst[2] = tmp;
306  }
307  break;
308  default:
309  LOG(FATAL) << "Invalid layout for 3d shape " << src_layout;
310  }
311  return dst;
312 }

322  Shape<4> dst;

341  case kNHWC:
342  dst2[0] = dst[0];

344  dst2[2] = dst[3];
345  dst2[3] = dst[1];
346  break;
347  default:

349  dst2 = src; // fixes compiler warning

486  inline void set_stream(Stream<Device> *stream) {
487  this->stream_ = stream;
488  }
493  template<int startdim>
494  MSHADOW_XINLINE index_t MemSize(void) const {
495  index_t memsz = this->stride_;
496  #pragma unroll
497  for (int i = startdim; i < kSubdim; ++i) {
498  memsz *= this->shape_[i];
499  }
500  return memsz;
501  }
506  MSHADOW_XINLINE bool CheckContiguous(void) const {
507  return this->shape_[dimension - 1] == stride_;
508  }
512  MSHADOW_XINLINE index_t MSize(void) const {
513  return this->MemSize<0>();
514  }
520  MSHADOW_XINLINE index_t size(int idx) const {
521  return shape_[idx];
522  }
527  MSHADOW_XINLINE Tensor<Device, 1, DType> FlatTo1D(void) const {
528  return Tensor<Device, 1, DType>(dptr_, shape_.FlatTo1D(), stride_, stream_);
529  }
534  MSHADOW_XINLINE Tensor<Device, 2, DType> FlatTo2D(void) const {
535  return Tensor<Device, 2, DType>(dptr_, shape_.FlatTo2D(), stride_, stream_);
536  }
542  MSHADOW_XINLINE Tensor<Device, kSubdim, DType> operator[](index_t idx) const {
543  return Tensor<Device, kSubdim, DType>(dptr_ + this->MemSize<1>() * idx,
544  shape_.SubShape(), stride_, stream_);
545  }
552  MSHADOW_XINLINE Tensor<Device, dimension, DType>
553  Slice(index_t begin, index_t end) const {
554  Shape<dimension> s = this->shape_;
555  s[0] = end - begin;
556  return Tensor<Device, dimension, DType>(dptr_ + this->MemSize<1>() * begin,
557  s, stride_, stream_);
558  }
560  inline Tensor<Device, dimension, DType> &
561  operator=(const Tensor<Device, dimension, DType> &exp) {
562  dptr_ = exp.dptr_;
563  shape_ = exp.shape_;

565  stream_ = exp.stream_;
566  return *this;
567  }

572  return this->__assign(exp);
573  }

576  return this->__assign(exp);
