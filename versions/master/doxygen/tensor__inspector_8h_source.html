256  void print_locator(const std::vector<index_t>& pos, std::vector<index_t>* sub_shape,
257  index_t* offset) {
258  const int dimension = tb_.ndim();
259  const int sub_dim = dimension - pos.size();
260  sub_shape->resize(sub_dim);
261  index_t multiple = 1;
262  for (size_t i = pos.size(), j = 0; i < static_cast<size_t>(dimension); ++i, ++j) {
263  (*sub_shape)[j] = tb_.shape_[i];
264  multiple *= tb_.shape_[i];
265  }
266  index_t sum = 0;
267  index_t m = 1;
268  for (index_t i = pos.size() - 1; i >= 0; --i) {
269  sum += pos[i] * m;
270  m *= tb_.shape_[i];
271  }
272  *offset = sum * multiple;
273  }

281  bool parse_position(std::vector<index_t>* pos, const std::string& str) {
282  const int dimension = tb_.ndim();
283  std::istringstream ss(str);
284  index_t n;
285  while (ss >> n) {
286  pos->push_back(n);
287  if (ss.peek() == ',') {
288  ss.ignore();
289  }
290  }
291  if (pos->size() > static_cast<size_t>(dimension)) {
292  return false;
293  }
294  for (size_t i = 0; i < pos->size(); ++i) {
295  if ((*pos)[i] > (tb_.shape_[i] - 1) || (*pos)[i] < 0) {
296  return false;
297  }
298  }
299  return !pos->empty();
300  }

371  template<typename DType>
372  std::function<bool(DType)> get_checker(CheckerType ct) {
373  switch (ct) {
374  case NegativeChecker:
375  return [] (DType x) {
376  return x < 0;
377  };
378  case PositiveChecker:
379  return [] (DType x) {
380  return x > 0;
381  };
382  case ZeroChecker:
383  return [] (DType x) {
384  return x == 0;
385  };
386  case NaNChecker:
387  if (std::is_same<DType, float>::value || std::is_same<DType, double>::value ||
388  std::is_same<DType, mshadow::half::half_t>::value) {
389  return [] (DType x) {
390  return x != x;
391  };
392  } else {
393  LOG(WARNING) << "NaNChecker only applies to float types. " <<
394  "Lambda will always return false.";
395  }
396  break;
397  case InfChecker:
398  if (std::is_same<DType, float>::value || std::is_same<DType, double>::value ||
399  std::is_same<DType, mshadow::half::half_t>::value) {
400  return [] (DType x) {
401  return x == (DType)1.0 / 0.0f || x == -(DType)1.0 / 0.0f;
402  };
403  } else {
404  LOG(WARNING) << "InfChecker only applies to float types. " <<
405  "Lambda will always return false.";
406  }
407  break;
408  case PositiveInfChecker:
409  if (std::is_same<DType, float>::value || std::is_same<DType, double>::value ||
410  std::is_same<DType, mshadow::half::half_t>::value) {
411  return [] (DType x) {
412  return x == (DType)1.0 / 0.0f;
413  };
414  } else {
415  LOG(WARNING) << "PositiveInfChecker only applies to float types. " <<
416  "Lambda will always return false.";
417  }
418  break;
419  case NegativeInfChecker:
420  if (std::is_same<DType, float>::value || std::is_same<DType, double>::value ||
421  std::is_same<DType, mshadow::half::half_t>::value) {
422  return [] (DType x) {
423  return x == -(DType)1.0 / 0.0f;
424  };
425  } else {
426  LOG(WARNING) << "NegativeInfChecker only applies to float types. " <<
427  "Lambda will always return false.";
428  }
429  break;
430  case FiniteChecker:
431  if (std::is_same<DType, float>::value || std::is_same<DType, double>::value ||
432  std::is_same<DType, mshadow::half::half_t>::value) {
433  return [] (DType x) {
434  return x != (DType)1.0 / 0.0f && x != -(DType)1.0 / 0.0f;
435  };
436  } else {
437  LOG(WARNING) << "FiniteChecker only applies to float types. " <<
438  "Lambda will always return false.";
439  }
440  break;
441  case NormalChecker:
442  if (std::is_same<DType, float>::value || std::is_same<DType, double>::value ||
443  std::is_same<DType, mshadow::half::half_t>::value) {
444  return [] (DType x) {
445  return x != (DType)1.0 / 0.0f && x != -(DType)1.0 / 0.0f &&
446  x == x;
447  };
448  } else {
449  LOG(WARNING) << "NormalChecker only applies to float types. " <<
450  "Lambda will always return false.";
451  }
452  break;
453  case AbnormalChecker:
454  if (std::is_same<DType, float>::value || std::is_same<DType, double>::value ||
455  std::is_same<DType, mshadow::half::half_t>::value) {
456  return [] (DType x) {
457  return x == (DType)1.0 / 0.0f || x == -(DType)1.0 / 0.0f ||
458  x != x;
459  };
460  } else {
461  LOG(WARNING) << "AbnormalChecker only applies to float types. " <<
462  "Lambda will always return false.";
463  }
464  break;
465  default:
466  return [] (DType x) {
467  return false;
468  };
469  }
470  return [] (DType x) {return false;};
471  }
