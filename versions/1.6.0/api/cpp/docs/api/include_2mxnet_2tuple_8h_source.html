271  friend std::istream &operator>>(std::istream &is, Tuple<ValueType> &t) {
272  // get (
273  while (true) {
274  char ch = is.peek();
275  if (isdigit(ch) || ch == '-') {
276  ValueType idx;
277  if (is >> idx) {
278  t.assign(&idx, &idx + 1);
279  }
280  return is;
281  }
282  is.get();
283  if (ch == '(' || ch == '[') break;
284  if (!isspace(ch)) {
285  if (ch == 'N') {
286  std::string tmp_val;
287  is >> tmp_val;
288  if (tmp_val == "one") { // is stores "None"
289  t.SetDim(-1);
290  return is;
291  }
292  }
293  is.setstate(std::ios::failbit);
294  return is;
295  }
296  }
297  // Handle empty tuple. A tensor whose shape is an empty tuple
298  // represents a scalar with ndim = 0.
299  while (isspace(is.peek())) {
300  is.get();
301  }
302  if (is.peek() == ')' || is.peek() == ']') {
303  is.get();
304  t.SetDim(0);
305  return is;
306  }
307  // Handle non-empty tuple
308  ValueType idx;
309  std::vector<ValueType> tmp;
310  while (is >> idx) {
311  tmp.push_back(idx);
312  char ch;
313  do {
314  ch = is.get();
315  } while (isspace(ch));
316  if (std::is_integral<ValueType>::value && ch == 'L') {
317  ch = is.get();
318  }
319  if (ch == ',') {
320  while (true) {
321  ch = is.peek();
322  if (isspace(ch)) {
323  is.get(); continue;
324  }
325  if (ch == ')' || ch == ']') {
326  is.get(); break;
327  }
328  break;
329  }
330  if (ch == ')' || ch == ']') break;
331  } else if (ch == ')' || ch == ']') {
332  break;
333  } else {
334  is.setstate(std::ios::failbit);
335  return is;

337  }
