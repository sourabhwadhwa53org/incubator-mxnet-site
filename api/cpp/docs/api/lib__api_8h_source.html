517 struct JsonVal {
518  JsonVal() : type(ERR), num(-1), str("") {} // default constructor
519  // construct a JSON object by type
520  explicit JsonVal(JsonType t) : type(t), num(-1), str("") {}
521  // construct a string JSON object
522  explicit JsonVal(std::string s) : type(STR), num(-1), str(s) {}
523  // construct a number JSON object
524  explicit JsonVal(int n) : type(NUM), num(n), str(std::to_string(n)) {}
525  // complex constructor
526  JsonVal(JsonType t, int n, std::string s) : type(t), num(n), str(s) {}
527  bool operator<(const JsonVal &o) const {
528  // for string JSON objects compare the string
529  if (type == STR) return type == o.type && str < o.str;
530  // for number JSON objects compare the number
531  if (type == NUM) return type == o.type && num < o.num;
532  // for list JSON objects, compare the size of list, and then each object in the list
533  if (type == LIST) {
534  if (list.size() != o.list.size()) return false;
535  for (unsigned int i=0; i< list.size(); i++)
536  if (list[i] < o.list[i])
537  return false; // if we find an object that doesnt match return
538  return true; // all objects in lists matched
539  }
540  // for map JSON objects, compare the size of map, and then each key/value in the maps
541  if (type == MAP) {
542  if (map.size() != o.map.size()) return false;
543  for (auto &item : map) {
544  // if one map is missing a key in another return
545  if (o.map.find(item.first) == o.map.end()) return false;
546  if (item.second < o.map.at(item.first)) return false;
547  }
548  return true;
549  }
550  return type < o.type;
551  }
552  JsonType type;
553  int num;
554  std::string str;
555  std::vector<JsonVal> list;
556  std::map<JsonVal, JsonVal> map;
557 };
