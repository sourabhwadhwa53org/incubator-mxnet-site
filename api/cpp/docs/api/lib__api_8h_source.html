327 struct JsonVal {
328  JsonVal() : type(ERR), num(-1), str("") {} // default constructor
329  // construct a JSON object by type
330  explicit JsonVal(JsonType t) : type(t), num(-1), str("") {}
331  // construct a string JSON object
332  explicit JsonVal(std::string s) : type(STR), num(-1), str(s) {}
333  // construct a number JSON object
334  explicit JsonVal(int n) : type(NUM), num(n), str(std::to_string(n)) {}
335  // complex constructor
336  JsonVal(JsonType t, int n, std::string s) : type(t), num(n), str(s) {}
337  bool operator<(const JsonVal &o) const {
338  // for string JSON objects compare the string
339  if (type == STR) return type == o.type && str < o.str;
340  // for number JSON objects compare the number
341  if (type == NUM) return type == o.type && num < o.num;
342  // for list JSON objects, compare the size of list, and then each object in the list
343  if (type == LIST) {
344  if (list.size() != o.list.size()) return false;
345  for (unsigned int i=0; i< list.size(); i++)
346  if (list[i] < o.list[i])
347  return false; // if we find an object that doesnt match return
348  return true; // all objects in lists matched
349  }
350  // for map JSON objects, compare the size of map, and then each key/value in the maps
351  if (type == MAP) {
352  if (map.size() != o.map.size()) return false;
353  for (auto &item : map) {
354  // if one map is missing a key in another return
355  if (o.map.find(item.first) == o.map.end()) return false;
356  if (item.second < o.map.at(item.first)) return false;
357  }
358  return true;
359  }
360  return type < o.type;
361  }
362  JsonType type;
363  int num;
364  std::string str;
365  std::vector<JsonVal> list;
366  std::map<JsonVal, JsonVal> map;
367 };

378  // debug function to convert a JSON object to a string
379  std::string json_val_string(const JsonVal &val) {
380  std::string ret;
381  switch (val.type) {
382  case ERR:
383  ret = "json(Error)";
384  break;
385  case STR:
386  ret = "json(STR:" + val.str + ")";
387  break;
388  case NUM:
389  ret = "json(INT:" + val.str + ")";
390  break;
391  case LIST:
392  ret = "json(LIST:[";
393  for (auto &item : val.list)
394  ret += json_val_string(item) + ",";
395  ret += "])";
396  break;
397  case MAP:
398  ret = "json(MAP:{";
399  for (auto &item : val.map)
400  ret += json_val_string(item.first) + " : " + json_val_string(item.second) + ",";
401  ret += "})";
402  break;
403  }
404  return ret;
405  }
406  // parse a string JSON object
407  JsonVal parse_string(std::string json, unsigned int* idx) {
408  JsonVal ret(STR);
409  while (*idx < json.size()) {
410  if (json[*idx] == '"') {
411  ++(*idx);
412  return ret;
413  } else {
414  ret.str += json[*idx];
415  ++(*idx);
416  }
417  }
418  std::cout << "Error! Unable to parse string" << std::endl;
419  return JsonVal();
420  }
421  // parse a number JSON object
422  JsonVal parse_num(std::string json, unsigned int* idx) {
423  JsonVal ret(NUM);
424  while (*idx < json.size()) {
425  if (json[*idx] >= '0' && json[*idx] <= '9') {
426  ret.str += json[*idx];
427  ++(*idx);
428  } else {
429  break;
430  }
431  }
432  ret.num = std::stoi(ret.str);
433  return ret;
434  }
435  // parse a list of JSON objects
436  JsonVal parse_list(std::string json, unsigned int* idx) {
437  JsonVal ret(LIST);
438  while (*idx < json.size()) {
439  if (json[*idx] == ']') {
440  ++(*idx);
441  return ret;
442  } else {
443  JsonVal item = parse(json, idx);
444  if (item.type != ERR)
445  ret.list.push_back(item);
446  }
447  }
448  std::cout << "Error! Unable to parse list" << std::endl;
449  return JsonVal();
450  }
451  // parse a map of JSON objects
452  JsonVal parse_map(std::string json, unsigned int* idx) {
453  JsonVal ret(MAP), key;
454  while (*idx < json.size()) {
455  if (json[*idx] == '}') {
456  ++(*idx);
457  return ret;
458  } else {
459  JsonVal item = parse(json, idx);
460  if (key.type == ERR) {
461  key = item;
462  } else {
463  ret.map[key] = item;
464  key.type = ERR;
465  }
466  }
467  }
468  std::cout << "Error! Unable to parse map" << std::endl;
469  return JsonVal();
470  }
471  // generic parse function
472  JsonVal parse(std::string json, unsigned int *idx) {
473  JsonVal ret;
474  while (*idx < json.size()) {
475  if (json[*idx] == '"') {
476  ++(*idx);
477  ret = parse_string(json, idx);
478  } else if (json[*idx] >= '0' && json[*idx] <= '9') {
479  ret = parse_num(json, idx);
480  } else if (json[*idx] == '[') {
481  ++(*idx);
482  ret = parse_list(json, idx);
483  } else if (json[*idx] == '{') {
484  ++(*idx);
485  ret = parse_map(json, idx);
486  } else if (json[*idx] == ']' || json[*idx] == '}') {return ret;}
487  if (ret.type != ERR) return ret;
488  ++(*idx);
489  }
490  return ret;
491  }
