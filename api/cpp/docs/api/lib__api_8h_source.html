415 struct JsonVal {
416  JsonVal() : type(ERR), num(-1), str("") {} // default constructor
417  // construct a JSON object by type
418  explicit JsonVal(JsonType t) : type(t), num(-1), str("") {}
419  // construct a string JSON object
420  explicit JsonVal(std::string s) : type(STR), num(-1), str(s) {}
421  // construct a number JSON object
422  explicit JsonVal(int n) : type(NUM), num(n), str(std::to_string(n)) {}
423  // complex constructor
424  JsonVal(JsonType t, int n, std::string s) : type(t), num(n), str(s) {}
425  bool operator<(const JsonVal &o) const {
426  // for string JSON objects compare the string
427  if (type == STR) return type == o.type && str < o.str;
428  // for number JSON objects compare the number
429  if (type == NUM) return type == o.type && num < o.num;
430  // for list JSON objects, compare the size of list, and then each object in the list
431  if (type == LIST) {
432  if (list.size() != o.list.size()) return false;
433  for (unsigned int i=0; i< list.size(); i++)
434  if (list[i] < o.list[i])
435  return false; // if we find an object that doesnt match return
436  return true; // all objects in lists matched
437  }
438  // for map JSON objects, compare the size of map, and then each key/value in the maps
439  if (type == MAP) {
440  if (map.size() != o.map.size()) return false;
441  for (auto &item : map) {
442  // if one map is missing a key in another return
443  if (o.map.find(item.first) == o.map.end()) return false;
444  if (item.second < o.map.at(item.first)) return false;
445  }
446  return true;
447  }
448  return type < o.type;
449  }
450  JsonType type;
451  int num;
452  std::string str;
453  std::vector<JsonVal> list;
454  std::map<JsonVal, JsonVal> map;
455 };

495  JsonVal parse_string(std::string json, unsigned int* idx) {
496  JsonVal ret(STR);
497  while (*idx < json.size()) {
498  if (json[*idx] == '"') {
499  ++(*idx);
500  return ret;

503  ++(*idx);
504  }
505  }

509  // parse a number JSON object
510  JsonVal parse_num(std::string json, unsigned int* idx) {
511  JsonVal ret(NUM);
512  while (*idx < json.size()) {
513  if (json[*idx] >= '0' && json[*idx] <= '9') {
514  ret.str += json[*idx];
515  ++(*idx);
516  } else {
517  break;
518  }
519  }
520  ret.num = std::stoi(ret.str);
521  return ret;
522  }
523  // parse a list of JSON objects
524  JsonVal parse_list(std::string json, unsigned int* idx) {
525  JsonVal ret(LIST);
526  while (*idx < json.size()) {
527  if (json[*idx] == ']') {
528  ++(*idx);
529  return ret;
530  } else {
531  JsonVal item = parse(json, idx);
532  if (item.type != ERR)

539  // parse a map of JSON objects
540  JsonVal parse_map(std::string json, unsigned int* idx) {
541  JsonVal ret(MAP), key;
542  while (*idx < json.size()) {
543  if (json[*idx] == '}') {
544  ++(*idx);
545  return ret;
546  } else {
547  JsonVal item = parse(json, idx);
548  if (key.type == ERR) {

558  }
559  // generic parse function
560  JsonVal parse(std::string json, unsigned int *idx) {
561  JsonVal ret;
562  while (*idx < json.size()) {
563  if (json[*idx] == '"') {
564  ++(*idx);
565  ret = parse_string(json, idx);
566  } else if (json[*idx] >= '0' && json[*idx] <= '9') {
567  ret = parse_num(json, idx);

571  } else if (json[*idx] == '{') {
572  ++(*idx);
573  ret = parse_map(json, idx);
574  } else if (json[*idx] == ']' || json[*idx] == '}') {return ret;}
575  if (ret.type != ERR) return ret;
576  ++(*idx);
577  }
