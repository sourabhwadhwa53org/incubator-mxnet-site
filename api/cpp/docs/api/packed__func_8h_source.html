820 inline DLDataType String2DLDataType(std::string s) {
821  DLDataType t;
822  // handle None type
823  if (s.length() == 0) {
824  t.bits = 0; t.lanes = 0; t.code = kHandle;
825  return t;
826  }
827  t.bits = 32; t.lanes = 1;
828  const char* scan = nullptr;
829  if (s.substr(0, 3) == "int") {
830  t.code = kDLInt; scan = s.c_str() + 3;
831  } else if (s.substr(0, 4) == "uint") {
832  t.code = kDLUInt; scan = s.c_str() + 4;
833  } else if (s.substr(0, 5) == "float") {
834  t.code = kDLFloat; scan = s.c_str() + 5;
835  } else if (s.substr(0, 6) == "handle") {
836  t.code = kHandle;
837  t.bits = 64; // handle uses 64 bit by default.
838  scan = s.c_str() + 6;
839  } else if (s == "bool") {
840  t.code = kDLUInt;
841  t.bits = 1;
842  t.lanes = 1;
843  return t;
844  } else if (s.substr(0, 6) == "custom") {
845  LOG(FATAL) << "custom MXNetDataType is not supported";
846  // t.code = ParseCustomDatatype(s, &scan);
847  } else {
848  scan = s.c_str();
849  LOG(FATAL) << "unknown type " << s;
850  }
851  char* xdelim; // emulate sscanf("%ux%u", bits, lanes)
852  uint8_t bits = static_cast<uint8_t>(strtoul(scan, &xdelim, 10));
853  if (bits != 0) t.bits = bits;
854  char* endpt = xdelim;
855  if (*xdelim == 'x') {
856  t.lanes = static_cast<uint16_t>(strtoul(xdelim + 1, &endpt, 10));
857  }
858  CHECK(endpt == s.c_str() + s.length()) << "unknown type " << s;
859  return t;
860 }
