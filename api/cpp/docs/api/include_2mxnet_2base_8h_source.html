232 class GPUAuxStream {
233  public:
238  explicit GPUAuxStream(mshadow::Stream<gpu> *primary_stream) :
239  primary_stream_(primary_stream),
240  aux_stream_(primary_stream),
241  gpu_stream_sync_event_(nullptr) {
242  if (Context::GetGPUStreamsPerWorker() >= 2) {
243  // Create auxiliary stream on the same device with the same properties as the primary stream
244  bool primary_has_blas_handle =
245  primary_stream->blas_handle_ownership_ == mshadow::Stream<gpu>::OwnHandle;
246  bool primary_has_dnn_handle =
247  primary_stream->dnn_handle_ownership_ == mshadow::Stream<gpu>::OwnHandle;
248  aux_stream_ = mshadow::NewStream<gpu>(primary_has_blas_handle,
249  primary_has_dnn_handle,
250  primary_stream->dev_id);
251  MSHADOW_CUDA_CALL(cudaEventCreateWithFlags(&gpu_stream_sync_event_, cudaEventDisableTiming));
252  }
253  }
255  ~GPUAuxStream() {
256  // If the aux_stream_ == primary_stream_, then we created no new streams to destroy.
257  if (aux_stream_ != primary_stream_) {
258  MSHADOW_CATCH_ERROR(mshadow::DeleteStream<gpu>(aux_stream_));
259  MSHADOW_CATCH_ERROR(cudaEventDestroy(gpu_stream_sync_event_));
260  }
261  }
265  void PreAuxStreamUseSync() {
266  // If the aux_stream_ == primary_stream_, then no synchronization is necessary.
267  if (aux_stream_ != primary_stream_)
268  StreamSync(primary_stream_, aux_stream_, gpu_stream_sync_event_);
269  }
273  void PostAuxStreamUseSync() {
274  // If the aux_stream_ == primary_stream_, then no synchronization is necessary.
275  if (aux_stream_ != primary_stream_)
276  StreamSync(aux_stream_, primary_stream_, gpu_stream_sync_event_);
277  }
279  mshadow::Stream<gpu> *GetStream() { return aux_stream_; }
286  static void StreamSync(mshadow::Stream<gpu> *s1, mshadow::Stream<gpu> *s2, cudaEvent_t event) {
287  MSHADOW_CUDA_CALL(cudaEventRecord(event, s1->stream_));
288  MSHADOW_CUDA_CALL(cudaStreamWaitEvent(s2->stream_, event, 0));
289  }
290 
291  private:
292  mshadow::Stream<gpu> *primary_stream_;
293  mshadow::Stream<gpu> *aux_stream_;
294  cudaEvent_t gpu_stream_sync_event_;
295 };

390 inline Context Context::Create(DeviceType dev_type, int32_t dev_id) {
391  Context ctx;
392  ctx.dev_type = dev_type;
393  ctx.dev_id = dev_id < 0 ? 0 : dev_id;
394  if (dev_type & kGPU) {
395 #if MXNET_USE_CUDA

398 #if MXNET_USE_CUDNN

400 #endif

404 #else
405  LOG(FATAL) << "Please compile with CUDA enabled for cuda features";
406 #endif
407  }
408  }
409  return ctx;
410 }

494 inline Context Context::FromString(const std::string& str) {
495  Context ret;
496  try {
497  const std::string::size_type l = str.find('(');
498  CHECK_NE(l, std::string::npos);
499  const std::string::size_type r = str.find(')');
500  CHECK_EQ(r, str.length()-1);
501 
502  const std::string type = str.substr(0, l);
503  int id = std::stoi(str.substr(l+1, r-l-1));
504  if (type == "cpu") {
505  ret = CPU(id);
506  } else if (type == "gpu") {
507  ret = GPU(id);
508  } else if (type == "cpu_pinned") {
509  ret = CPUPinned(id);
510  } else if (type == "cpu_shared") {
511  ret = CPUShared(id);
512  } else {
513  LOG(FATAL) << "Invalid context string " << str;
514  }
515  } catch (...) {
516  LOG(FATAL) << "Invalid context string " << str;
517  }
518  return ret;
519 }
