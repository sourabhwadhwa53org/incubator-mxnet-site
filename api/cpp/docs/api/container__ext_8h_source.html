754 inline int String::memncmp(const char* lhs, const char* rhs, size_t lhs_count, size_t rhs_count) {
755  if (lhs == rhs && lhs_count == rhs_count) return 0;
756 
757  for (size_t i = 0; i < lhs_count && i < rhs_count; ++i) {
758  if (lhs[i] < rhs[i]) return -1;
759  if (lhs[i] > rhs[i]) return 1;
760  }
761  if (lhs_count < rhs_count) {
762  return -1;
763  } else if (lhs_count > rhs_count) {
764  return 1;
765  } else {
766  return 0;
767  }
768 }
